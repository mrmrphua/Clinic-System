	DISID   	VARCHAR(10) NOT NULL,
	DISNAME 	VARCHAR(30),
	UNITPRICE	VARCHAR(7),
	QUANTITY	VARCHAR(7),
	USAGE		VARCHAR(60),

PRIMARY KEY(DISID));

CREATE TABLE PAYMENT(
	PAYMENTCODE	VARCHAR(10) NOT NULL,
	PAT_ID		VARCHAR(10),
        STAFF_ID        VARCHAR(10),
	AMOUNT		DOUBLE,

PRIMARY KEY(PAYMENTCODE),
FOREIGN KEY (PAT_ID) REFERENCES PATIENT(PATID),
FOREIGN KEY (STAFF_ID) REFERENCES STAFF(STAFFID));
